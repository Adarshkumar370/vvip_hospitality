-- Invoices raised against a user, normally for an order.
-- order_id links invoices to customer orders.

CREATE SEQUENCE IF NOT EXISTS invoice_number_seq
    START WITH 1
    INCREMENT BY 1;

CREATE TABLE IF NOT EXISTS invoices (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    invoice_number varchar(40) NOT NULL DEFAULT ('INV-' || lpad(nextval('invoice_number_seq')::text, 10, '0')),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE RESTRICT,
    order_id uuid REFERENCES orders(id) ON DELETE RESTRICT,
    invoice_status invoice_status NOT NULL DEFAULT 'draft',
    currency_code char(3) NOT NULL DEFAULT 'INR',
    subtotal_amount numeric(12, 2) NOT NULL DEFAULT 0,
    discount_amount numeric(12, 2) NOT NULL DEFAULT 0,
    tax_amount numeric(12, 2) NOT NULL DEFAULT 0,
    shipping_amount numeric(12, 2) NOT NULL DEFAULT 0,
    adjustment_amount numeric(12, 2) NOT NULL DEFAULT 0,
    total_amount numeric(12, 2) NOT NULL DEFAULT 0,
    paid_amount numeric(12, 2) NOT NULL DEFAULT 0,
    due_amount numeric(12, 2) NOT NULL DEFAULT 0,
    issued_at timestamptz,
    due_at timestamptz,
    paid_at timestamptz,
    voided_at timestamptz,
    cancelled_at timestamptz,
    billing_name varchar(150),
    billing_email citext,
    billing_mobile_no varchar(20),
    billing_address_snapshot jsonb,
    notes text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_invoices_number
        CHECK (length(btrim(invoice_number)) > 0),
    CONSTRAINT chk_invoices_currency
        CHECK (currency_code ~ '^[A-Z]{3}$'),
    CONSTRAINT chk_invoices_non_negative_amounts
        CHECK (
            subtotal_amount >= 0
            AND discount_amount >= 0
            AND tax_amount >= 0
            AND shipping_amount >= 0
            AND total_amount >= 0
            AND paid_amount >= 0
            AND due_amount >= 0
        ),
    CONSTRAINT chk_invoices_billing_name
        CHECK (billing_name IS NULL OR length(btrim(billing_name)) > 0),
    CONSTRAINT chk_invoices_billing_email
        CHECK (billing_email IS NULL OR length(btrim(billing_email::text)) > 0),
    CONSTRAINT chk_invoices_billing_mobile_no
        CHECK (
            billing_mobile_no IS NULL
            OR billing_mobile_no ~ '^\+[1-9][0-9]{7,14}$'
        ),
    CONSTRAINT chk_invoices_notes
        CHECK (notes IS NULL OR length(btrim(notes)) > 0),
    CONSTRAINT chk_invoices_issued_at
        CHECK (
            (invoice_status IN ('issued', 'partially_paid', 'paid', 'overdue') AND issued_at IS NOT NULL)
            OR invoice_status IN ('draft', 'void', 'cancelled')
        ),
    CONSTRAINT chk_invoices_paid_at
        CHECK (
            (invoice_status = 'paid' AND paid_at IS NOT NULL)
            OR invoice_status <> 'paid'
        ),
    CONSTRAINT chk_invoices_voided_at
        CHECK (
            (invoice_status = 'void' AND voided_at IS NOT NULL)
            OR invoice_status <> 'void'
        ),
    CONSTRAINT chk_invoices_cancelled_at
        CHECK (
            (invoice_status = 'cancelled' AND cancelled_at IS NOT NULL)
            OR invoice_status <> 'cancelled'
        )
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_invoices_invoice_number
    ON invoices (invoice_number);

CREATE INDEX IF NOT EXISTS ix_invoices_user_id
    ON invoices (user_id, created_at DESC);

CREATE INDEX IF NOT EXISTS ix_invoices_order_id
    ON invoices (order_id)
    WHERE order_id IS NOT NULL;

CREATE INDEX IF NOT EXISTS ix_invoices_payment_status
    ON invoices (invoice_status, due_at)
    WHERE invoice_status IN ('issued', 'partially_paid', 'overdue');

CREATE OR REPLACE FUNCTION invoices_prepare_amounts()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.total_amount := round(
        NEW.subtotal_amount
        - NEW.discount_amount
        + NEW.tax_amount
        + NEW.shipping_amount
        + NEW.adjustment_amount,
        2
    );

    IF NEW.total_amount < 0 THEN
        RAISE EXCEPTION 'invoice total cannot be negative';
    END IF;

    IF NEW.paid_amount > NEW.total_amount THEN
        RAISE EXCEPTION 'invoice paid amount cannot exceed total amount';
    END IF;

    NEW.due_amount := round(NEW.total_amount - NEW.paid_amount, 2);

    IF NEW.invoice_status IN ('issued', 'partially_paid', 'paid', 'overdue')
        AND NEW.issued_at IS NULL
    THEN
        NEW.issued_at := now();
    END IF;

    IF NEW.invoice_status NOT IN ('draft', 'void', 'cancelled') THEN
        IF NEW.total_amount > 0 AND NEW.paid_amount = NEW.total_amount THEN
            NEW.invoice_status := 'paid';
            NEW.paid_at := COALESCE(NEW.paid_at, now());
        ELSIF NEW.paid_amount > 0 THEN
            NEW.invoice_status := 'partially_paid';
            NEW.paid_at := NULL;
        ELSIF NEW.invoice_status IN ('paid', 'partially_paid') THEN
            NEW.invoice_status := 'issued';
            NEW.paid_at := NULL;
        END IF;
    END IF;

    IF NEW.invoice_status = 'void' THEN
        NEW.voided_at := COALESCE(NEW.voided_at, now());
    ELSIF NEW.invoice_status = 'cancelled' THEN
        NEW.cancelled_at := COALESCE(NEW.cancelled_at, now());
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION invoices_prevent_delete()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'invoices cannot be deleted; void or cancel the invoice instead';
END;
$$;

DROP TRIGGER IF EXISTS trg_invoices_prepare_amounts ON invoices;

CREATE TRIGGER trg_invoices_prepare_amounts
BEFORE INSERT OR UPDATE OF invoice_status, subtotal_amount, discount_amount, tax_amount, shipping_amount, adjustment_amount, paid_amount
ON invoices
FOR EACH ROW
EXECUTE FUNCTION invoices_prepare_amounts();

DROP TRIGGER IF EXISTS trg_invoices_set_updated_at ON invoices;

CREATE TRIGGER trg_invoices_set_updated_at
BEFORE UPDATE ON invoices
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_invoices_prevent_delete ON invoices;

CREATE TRIGGER trg_invoices_prevent_delete
BEFORE DELETE ON invoices
FOR EACH ROW
EXECUTE FUNCTION invoices_prevent_delete();

DROP TRIGGER IF EXISTS trg_invoices_outbox ON invoices;

CREATE TRIGGER trg_invoices_outbox
AFTER INSERT OR UPDATE OR DELETE ON invoices
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
