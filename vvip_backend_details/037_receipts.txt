-- Receipts issued for successful payments.
-- A receipt is proof that money was received. Pending and failed attempts stay
-- in payments and must not get receipt rows.

CREATE SEQUENCE IF NOT EXISTS receipt_number_seq
    START WITH 1
    INCREMENT BY 1;

CREATE TABLE IF NOT EXISTS receipts (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    receipt_number varchar(40) NOT NULL DEFAULT ('RCT-' || lpad(nextval('receipt_number_seq')::text, 10, '0')),
    payment_id uuid NOT NULL UNIQUE REFERENCES payments(id) ON DELETE RESTRICT,
    invoice_id uuid REFERENCES invoices(id) ON DELETE RESTRICT,
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE RESTRICT,
    amount numeric(12, 2) NOT NULL,
    currency_code char(3) NOT NULL DEFAULT 'INR',
    receipt_status receipt_status NOT NULL DEFAULT 'issued',
    issued_at timestamptz NOT NULL DEFAULT now(),
    voided_at timestamptz,
    void_reason text,
    notes text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_receipts_number
        CHECK (length(btrim(receipt_number)) > 0),
    CONSTRAINT chk_receipts_amount
        CHECK (amount > 0),
    CONSTRAINT chk_receipts_currency
        CHECK (currency_code ~ '^[A-Z]{3}$'),
    CONSTRAINT chk_receipts_voided_at
        CHECK (
            (receipt_status = 'void' AND voided_at IS NOT NULL)
            OR receipt_status <> 'void'
        ),
    CONSTRAINT chk_receipts_void_reason
        CHECK (void_reason IS NULL OR length(btrim(void_reason)) > 0),
    CONSTRAINT chk_receipts_notes
        CHECK (notes IS NULL OR length(btrim(notes)) > 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_receipts_receipt_number
    ON receipts (receipt_number);

CREATE INDEX IF NOT EXISTS ix_receipts_invoice_id
    ON receipts (invoice_id, issued_at DESC)
    WHERE invoice_id IS NOT NULL;

CREATE INDEX IF NOT EXISTS ix_receipts_user_id
    ON receipts (user_id, issued_at DESC);

CREATE OR REPLACE FUNCTION receipts_validate_payment()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_payment payments%ROWTYPE;
BEGIN
    SELECT *
    INTO v_payment
    FROM payments
    WHERE id = NEW.payment_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'payment % does not exist', NEW.payment_id;
    END IF;

    IF v_payment.payment_status <> 'succeeded' THEN
        RAISE EXCEPTION 'receipt requires a succeeded payment';
    END IF;

    NEW.user_id := v_payment.user_id;
    NEW.invoice_id := v_payment.invoice_id;
    NEW.amount := v_payment.amount;
    NEW.currency_code := v_payment.currency_code;

    IF NEW.receipt_status = 'void' THEN
        NEW.voided_at := COALESCE(NEW.voided_at, now());
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION receipts_prevent_delete()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'receipts cannot be deleted; void the receipt instead';
END;
$$;

DROP TRIGGER IF EXISTS trg_receipts_validate_payment ON receipts;

CREATE TRIGGER trg_receipts_validate_payment
BEFORE INSERT OR UPDATE OF payment_id, receipt_status ON receipts
FOR EACH ROW
EXECUTE FUNCTION receipts_validate_payment();

DROP TRIGGER IF EXISTS trg_receipts_set_updated_at ON receipts;

CREATE TRIGGER trg_receipts_set_updated_at
BEFORE UPDATE ON receipts
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_receipts_prevent_delete ON receipts;

CREATE TRIGGER trg_receipts_prevent_delete
BEFORE DELETE ON receipts
FOR EACH ROW
EXECUTE FUNCTION receipts_prevent_delete();

DROP TRIGGER IF EXISTS trg_receipts_outbox ON receipts;

CREATE TRIGGER trg_receipts_outbox
AFTER INSERT OR UPDATE OR DELETE ON receipts
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
