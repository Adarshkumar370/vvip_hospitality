-- Staff work orders.
-- These are internal tasks created from customer orders. A staff member can
-- mark their assigned task complete; admins or staff with complete_work_orders
-- can complete tasks for someone else.

CREATE SEQUENCE IF NOT EXISTS work_order_number_seq
    START WITH 1
    INCREMENT BY 1;

CREATE TABLE IF NOT EXISTS work_orders (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    work_order_number varchar(40) NOT NULL DEFAULT ('WO-' || lpad(nextval('work_order_number_seq')::text, 10, '0')),
    order_id uuid NOT NULL REFERENCES orders(id) ON DELETE RESTRICT,
    order_item_id uuid REFERENCES order_items(id) ON DELETE SET NULL,
    task_type work_order_task_type NOT NULL,
    work_status work_order_status NOT NULL DEFAULT 'pending',
    assigned_designation staff_designation,
    assigned_staff_id uuid REFERENCES staff_members(id) ON DELETE SET NULL,
    created_by_staff_id uuid REFERENCES staff_members(id) ON DELETE SET NULL,
    completed_by_staff_id uuid REFERENCES staff_members(id) ON DELETE SET NULL,
    priority smallint NOT NULL DEFAULT 3,
    title varchar(160) NOT NULL,
    instructions text,
    due_at timestamptz,
    assigned_at timestamptz,
    started_at timestamptz,
    completed_at timestamptz,
    cancelled_at timestamptz,
    completion_notes text,
    blocked_reason text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_work_orders_number
        CHECK (length(btrim(work_order_number)) > 0),
    CONSTRAINT chk_work_orders_assignment_target
        CHECK (assigned_staff_id IS NOT NULL OR assigned_designation IS NOT NULL),
    CONSTRAINT chk_work_orders_priority
        CHECK (priority BETWEEN 1 AND 5),
    CONSTRAINT chk_work_orders_title
        CHECK (length(btrim(title)) > 0),
    CONSTRAINT chk_work_orders_instructions
        CHECK (instructions IS NULL OR length(btrim(instructions)) > 0),
    CONSTRAINT chk_work_orders_completion_notes
        CHECK (completion_notes IS NULL OR length(btrim(completion_notes)) > 0),
    CONSTRAINT chk_work_orders_blocked_reason
        CHECK (blocked_reason IS NULL OR length(btrim(blocked_reason)) > 0),
    CONSTRAINT chk_work_orders_completed_at
        CHECK (
            (work_status = 'completed' AND completed_at IS NOT NULL AND completed_by_staff_id IS NOT NULL)
            OR work_status <> 'completed'
        ),
    CONSTRAINT chk_work_orders_cancelled_at
        CHECK (
            (work_status = 'cancelled' AND cancelled_at IS NOT NULL)
            OR work_status <> 'cancelled'
        ),
    CONSTRAINT chk_work_orders_blocked
        CHECK (
            (work_status = 'blocked' AND blocked_reason IS NOT NULL)
            OR work_status <> 'blocked'
        )
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_work_orders_number
    ON work_orders (work_order_number);

CREATE INDEX IF NOT EXISTS ix_work_orders_order_id
    ON work_orders (order_id, work_status);

CREATE INDEX IF NOT EXISTS ix_work_orders_assigned_staff
    ON work_orders (assigned_staff_id, work_status, due_at)
    WHERE assigned_staff_id IS NOT NULL;

CREATE INDEX IF NOT EXISTS ix_work_orders_assigned_designation
    ON work_orders (assigned_designation, work_status, due_at)
    WHERE assigned_designation IS NOT NULL;

CREATE OR REPLACE FUNCTION work_orders_prepare_and_validate()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_assigned_staff staff_members%ROWTYPE;
    v_created_by staff_members%ROWTYPE;
    v_completed_by staff_members%ROWTYPE;
BEGIN
    IF NEW.created_by_staff_id IS NOT NULL THEN
        SELECT *
        INTO v_created_by
        FROM staff_members
        WHERE id = NEW.created_by_staff_id;

        IF NOT FOUND OR v_created_by.status <> 'active' THEN
            RAISE EXCEPTION 'created_by_staff_id must be active staff';
        END IF;

        IF NOT staff_has_permission(NEW.created_by_staff_id, 'assign_work_orders') THEN
            RAISE EXCEPTION 'staff member does not have permission to assign work orders';
        END IF;
    END IF;

    IF NEW.order_item_id IS NOT NULL
        AND NOT EXISTS (
            SELECT 1
            FROM order_items oi
            WHERE oi.id = NEW.order_item_id
              AND oi.order_id = NEW.order_id
        )
    THEN
        RAISE EXCEPTION 'order_item_id must belong to order_id';
    END IF;

    IF NEW.assigned_staff_id IS NOT NULL THEN
        SELECT *
        INTO v_assigned_staff
        FROM staff_members
        WHERE id = NEW.assigned_staff_id;

        IF NOT FOUND OR v_assigned_staff.status <> 'active' THEN
            RAISE EXCEPTION 'assigned staff must be active';
        END IF;

        IF NEW.assigned_designation IS NULL THEN
            NEW.assigned_designation := v_assigned_staff.designation;
        ELSIF v_assigned_staff.designation <> NEW.assigned_designation
            AND v_assigned_staff.designation <> 'admin'
        THEN
            RAISE EXCEPTION 'assigned staff designation does not match work order designation';
        END IF;

        NEW.assigned_at := COALESCE(NEW.assigned_at, now());
        IF NEW.work_status = 'pending' THEN
            NEW.work_status := 'assigned';
        END IF;
    END IF;

    IF NEW.work_status = 'in_progress' THEN
        NEW.started_at := COALESCE(NEW.started_at, now());
    ELSIF NEW.work_status = 'completed' THEN
        NEW.completed_at := COALESCE(NEW.completed_at, now());

        SELECT *
        INTO v_completed_by
        FROM staff_members
        WHERE id = NEW.completed_by_staff_id;

        IF NOT FOUND OR v_completed_by.status <> 'active' THEN
            RAISE EXCEPTION 'completed_by_staff_id must be active staff';
        END IF;

        IF NEW.assigned_staff_id IS NOT NULL
            AND NEW.assigned_staff_id IS DISTINCT FROM NEW.completed_by_staff_id
            AND NOT staff_has_permission(NEW.completed_by_staff_id, 'complete_work_orders')
        THEN
            RAISE EXCEPTION 'staff cannot complete work order assigned to another staff member';
        END IF;

        IF NEW.assigned_staff_id IS NULL
            AND v_completed_by.designation <> NEW.assigned_designation
            AND v_completed_by.designation <> 'admin'
            AND NOT staff_has_permission(NEW.completed_by_staff_id, 'complete_work_orders')
        THEN
            RAISE EXCEPTION 'staff cannot complete work order assigned to another designation';
        END IF;
    ELSIF NEW.work_status = 'cancelled' THEN
        NEW.cancelled_at := COALESCE(NEW.cancelled_at, now());
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION work_orders_prevent_delete()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'work orders cannot be deleted; cancel the work order instead';
END;
$$;

DROP TRIGGER IF EXISTS trg_work_orders_prepare_and_validate ON work_orders;

CREATE TRIGGER trg_work_orders_prepare_and_validate
BEFORE INSERT OR UPDATE OF order_id, order_item_id, assigned_designation, assigned_staff_id, created_by_staff_id, work_status, completed_by_staff_id
ON work_orders
FOR EACH ROW
EXECUTE FUNCTION work_orders_prepare_and_validate();

DROP TRIGGER IF EXISTS trg_work_orders_set_updated_at ON work_orders;

CREATE TRIGGER trg_work_orders_set_updated_at
BEFORE UPDATE ON work_orders
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_work_orders_prevent_delete ON work_orders;

CREATE TRIGGER trg_work_orders_prevent_delete
BEFORE DELETE ON work_orders
FOR EACH ROW
EXECUTE FUNCTION work_orders_prevent_delete();

DROP TRIGGER IF EXISTS trg_work_orders_outbox ON work_orders;

CREATE TRIGGER trg_work_orders_outbox
AFTER INSERT OR UPDATE ON work_orders
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
