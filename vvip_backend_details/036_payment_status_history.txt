-- Immutable payment status history.
-- This preserves pending, failed, retry, and success transitions.

CREATE TABLE IF NOT EXISTS payment_status_history (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    payment_id uuid NOT NULL REFERENCES payments(id) ON DELETE CASCADE,
    old_status payment_status,
    new_status payment_status NOT NULL,
    provider_status varchar(80),
    provider_status_message text,
    failure_code varchar(120),
    failure_message text,
    provider_response_payload jsonb,
    changed_by_actor_type varchar(30) NOT NULL DEFAULT 'system',
    changed_by_actor_id uuid,
    request_id uuid,
    created_at timestamptz NOT NULL DEFAULT now(),

    CONSTRAINT chk_payment_status_history_provider_status
        CHECK (provider_status IS NULL OR length(btrim(provider_status)) > 0),
    CONSTRAINT chk_payment_status_history_provider_message
        CHECK (provider_status_message IS NULL OR length(btrim(provider_status_message)) > 0),
    CONSTRAINT chk_payment_status_history_failure_code
        CHECK (failure_code IS NULL OR length(btrim(failure_code)) > 0),
    CONSTRAINT chk_payment_status_history_failure_message
        CHECK (failure_message IS NULL OR length(btrim(failure_message)) > 0),
    CONSTRAINT chk_payment_status_history_actor_type
        CHECK (changed_by_actor_type IN ('system', 'employee', 'provider'))
);

CREATE INDEX IF NOT EXISTS ix_payment_status_history_payment_id
    ON payment_status_history (payment_id, created_at DESC);

CREATE OR REPLACE FUNCTION payment_status_history_prevent_change()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'payment_status_history is append-only';
END;
$$;

CREATE OR REPLACE FUNCTION payments_write_status_history()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_actor_type text;
    v_actor_id_text text;
    v_actor_id uuid;
    v_request_id_text text;
    v_request_id uuid;
BEGIN
    IF TG_OP = 'UPDATE' AND NEW.payment_status IS NOT DISTINCT FROM OLD.payment_status THEN
        RETURN NEW;
    END IF;

    v_actor_type := COALESCE(NULLIF(current_setting('app.actor_type', true), ''), 'system');
    v_actor_id_text := NULLIF(current_setting('app.actor_id', true), '');
    v_request_id_text := NULLIF(current_setting('app.request_id', true), '');

    IF v_actor_id_text IS NOT NULL THEN
        v_actor_id := v_actor_id_text::uuid;
    END IF;

    IF v_request_id_text IS NOT NULL THEN
        v_request_id := v_request_id_text::uuid;
    END IF;

    INSERT INTO payment_status_history (
        payment_id,
        old_status,
        new_status,
        provider_status,
        provider_status_message,
        failure_code,
        failure_message,
        provider_response_payload,
        changed_by_actor_type,
        changed_by_actor_id,
        request_id
    )
    VALUES (
        NEW.id,
        CASE WHEN TG_OP = 'UPDATE' THEN OLD.payment_status ELSE NULL END,
        NEW.payment_status,
        NEW.provider_status,
        NEW.provider_status_message,
        NEW.failure_code,
        NEW.failure_message,
        NEW.provider_response_payload,
        v_actor_type,
        v_actor_id,
        v_request_id
    );

    RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_payments_write_status_history ON payments;

CREATE TRIGGER trg_payments_write_status_history
AFTER INSERT OR UPDATE OF payment_status ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_write_status_history();

DROP TRIGGER IF EXISTS trg_payment_status_history_prevent_change ON payment_status_history;

CREATE TRIGGER trg_payment_status_history_prevent_change
BEFORE UPDATE OR DELETE ON payment_status_history
FOR EACH ROW
EXECUTE FUNCTION payment_status_history_prevent_change();

DROP TRIGGER IF EXISTS trg_payment_status_history_outbox ON payment_status_history;

CREATE TRIGGER trg_payment_status_history_outbox
AFTER INSERT ON payment_status_history
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
