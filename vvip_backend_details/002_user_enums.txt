-- Enum definitions used by the user module.
-- DO blocks keep the script idempotent across repeated executions.

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'user_status') THEN
        CREATE TYPE user_status AS ENUM ('allowed', 'banned');
    END IF;
END
$$;

BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'user_payment_type') THEN

    END IF;
END
$$;
