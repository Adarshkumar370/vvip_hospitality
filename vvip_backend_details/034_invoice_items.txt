-- Invoice line items.
-- These rows snapshot product name, unit, and price so invoices remain correct
-- after catalog changes.

CREATE TABLE IF NOT EXISTS invoice_items (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    invoice_id uuid NOT NULL REFERENCES invoices(id) ON DELETE CASCADE,
    product_id uuid REFERENCES products(id) ON DELETE SET NULL,
    order_item_id uuid REFERENCES order_items(id) ON DELETE SET NULL,
    line_number integer NOT NULL DEFAULT 1,
    item_name text NOT NULL,
    item_description text,
    unit_code varchar(30) NOT NULL,
    unit_symbol varchar(20) NOT NULL,
    quantity numeric(12, 3) NOT NULL,
    unit_price numeric(12, 2) NOT NULL,
    line_subtotal numeric(12, 2) GENERATED ALWAYS AS (round(quantity * unit_price, 2)) STORED,
    discount_amount numeric(12, 2) NOT NULL DEFAULT 0,
    tax_amount numeric(12, 2) NOT NULL DEFAULT 0,
    line_total numeric(12, 2) GENERATED ALWAYS AS (
        round((quantity * unit_price) - discount_amount + tax_amount, 2)
    ) STORED,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_invoice_items_line_number
        CHECK (line_number > 0),
    CONSTRAINT chk_invoice_items_name
        CHECK (length(btrim(item_name)) > 0),
    CONSTRAINT chk_invoice_items_description
        CHECK (item_description IS NULL OR length(btrim(item_description)) > 0),
    CONSTRAINT chk_invoice_items_unit_code
        CHECK (unit_code ~ '^[a-z][a-z0-9_]{1,29}$'),
    CONSTRAINT chk_invoice_items_unit_symbol
        CHECK (length(btrim(unit_symbol)) > 0),
    CONSTRAINT chk_invoice_items_quantity
        CHECK (quantity > 0),
    CONSTRAINT chk_invoice_items_unit_price
        CHECK (unit_price >= 0),
    CONSTRAINT chk_invoice_items_amounts
        CHECK (discount_amount >= 0 AND tax_amount >= 0 AND line_total >= 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_invoice_items_invoice_line
    ON invoice_items (invoice_id, line_number);

CREATE INDEX IF NOT EXISTS ix_invoice_items_invoice_id
    ON invoice_items (invoice_id);

CREATE INDEX IF NOT EXISTS ix_invoice_items_product_id
    ON invoice_items (product_id)
    WHERE product_id IS NOT NULL;

CREATE OR REPLACE FUNCTION recalculate_invoice_totals(p_invoice_id uuid)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
    UPDATE invoices
    SET subtotal_amount = COALESCE((
            SELECT round(sum(line_subtotal), 2)
            FROM invoice_items
            WHERE invoice_id = p_invoice_id
        ), 0),
        discount_amount = COALESCE((
            SELECT round(sum(discount_amount), 2)
            FROM invoice_items
            WHERE invoice_id = p_invoice_id
        ), 0),
        tax_amount = COALESCE((
            SELECT round(sum(tax_amount), 2)
            FROM invoice_items
            WHERE invoice_id = p_invoice_id
        ), 0)
    WHERE id = p_invoice_id;
END;
$$;

CREATE OR REPLACE FUNCTION invoice_items_recalculate_invoice_totals()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    IF TG_OP = 'UPDATE' AND NEW.invoice_id IS DISTINCT FROM OLD.invoice_id THEN
        PERFORM recalculate_invoice_totals(OLD.invoice_id);
        PERFORM recalculate_invoice_totals(NEW.invoice_id);
        RETURN NEW;
    END IF;

    PERFORM recalculate_invoice_totals(COALESCE(NEW.invoice_id, OLD.invoice_id));
    RETURN COALESCE(NEW, OLD);
END;
$$;

DROP TRIGGER IF EXISTS trg_invoice_items_recalculate_invoice_totals ON invoice_items;

CREATE TRIGGER trg_invoice_items_recalculate_invoice_totals
AFTER INSERT OR UPDATE OR DELETE ON invoice_items
FOR EACH ROW
EXECUTE FUNCTION invoice_items_recalculate_invoice_totals();

DROP TRIGGER IF EXISTS trg_invoice_items_set_updated_at ON invoice_items;

CREATE TRIGGER trg_invoice_items_set_updated_at
BEFORE UPDATE ON invoice_items
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_invoice_items_outbox ON invoice_items;

CREATE TRIGGER trg_invoice_items_outbox
AFTER INSERT OR UPDATE OR DELETE ON invoice_items
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
