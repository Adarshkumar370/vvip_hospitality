-- Optional flexible metadata master table.
-- Use this only for non-core user metadata.

CREATE TABLE IF NOT EXISTS user_attribute_definitions (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    code varchar(80) NOT NULL,
    display_name varchar(120) NOT NULL,
    data_type attribute_data_type NOT NULL,
    allows_multiple boolean NOT NULL DEFAULT false,
    is_active boolean NOT NULL DEFAULT true,
    description text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_user_attribute_definitions_code
        CHECK (code ~ '^[a-z][a-z0-9_]{1,79}$'),
    CONSTRAINT chk_user_attribute_definitions_display_name
        CHECK (length(btrim(display_name)) > 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_user_attribute_definitions_code
    ON user_attribute_definitions (code);

DROP TRIGGER IF EXISTS trg_user_attribute_definitions_set_updated_at ON user_attribute_definitions;

CREATE TRIGGER trg_user_attribute_definitions_set_updated_at
BEFORE UPDATE ON user_attribute_definitions
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_user_attribute_definitions_outbox ON user_attribute_definitions;

CREATE TRIGGER trg_user_attribute_definitions_outbox
AFTER INSERT OR UPDATE OR DELETE ON user_attribute_definitions
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
