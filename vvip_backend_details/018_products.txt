-- Product catalog.
-- price is the current selling price per measurement_units row referenced by unit_id.
-- Orders should snapshot product name, unit, and price into order_items later.

CREATE TABLE IF NOT EXISTS products (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    category_id uuid NOT NULL REFERENCES product_categories(id) ON DELETE RESTRICT,
    unit_id uuid NOT NULL REFERENCES measurement_units(id) ON DELETE RESTRICT,
    name citext NOT NULL,
    slug varchar(160) NOT NULL,
    sku varchar(80),
    image_url text,
    price numeric(12, 2) NOT NULL,
    short_description text,
    long_description text,
    status catalog_visibility_status NOT NULL DEFAULT 'show',
    max_daily_limit numeric(12, 3) NOT NULL DEFAULT 100,
    min_order_quantity numeric(12, 3) NOT NULL DEFAULT 1,
    max_order_quantity numeric(12, 3),
    quantity_step numeric(12, 3) NOT NULL DEFAULT 1,
    display_order integer NOT NULL DEFAULT 0,
    is_customizable boolean NOT NULL DEFAULT false,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_products_name
        CHECK (length(btrim(name::text)) > 0),
    CONSTRAINT chk_products_slug
        CHECK (slug ~ '^[a-z0-9]+(?:-[a-z0-9]+)*$'),
    CONSTRAINT chk_products_sku
        CHECK (sku IS NULL OR length(btrim(sku)) > 0),
    CONSTRAINT chk_products_image_url
        CHECK (image_url IS NULL OR length(btrim(image_url)) > 0),
    CONSTRAINT chk_products_price
        CHECK (price >= 0),
    CONSTRAINT chk_products_short_description
        CHECK (short_description IS NULL OR length(btrim(short_description)) > 0),
    CONSTRAINT chk_products_long_description
        CHECK (long_description IS NULL OR length(btrim(long_description)) > 0),
    CONSTRAINT chk_products_max_daily_limit
        CHECK (max_daily_limit IS NULL OR max_daily_limit >= 0),
    CONSTRAINT chk_products_min_order_quantity
        CHECK (min_order_quantity > 0),
    CONSTRAINT chk_products_max_order_quantity
        CHECK (max_order_quantity IS NULL OR max_order_quantity >= min_order_quantity),
    CONSTRAINT chk_products_quantity_step
        CHECK (quantity_step > 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_products_slug
    ON products (slug);

CREATE UNIQUE INDEX IF NOT EXISTS ux_products_sku
    ON products (sku)
    WHERE sku IS NOT NULL;

CREATE UNIQUE INDEX IF NOT EXISTS ux_products_category_name
    ON products (category_id, name);

CREATE INDEX IF NOT EXISTS ix_products_visual_filter
    ON products (status, category_id, display_order, name);

CREATE INDEX IF NOT EXISTS ix_products_category_id
    ON products (category_id);

CREATE INDEX IF NOT EXISTS ix_products_unit_id
    ON products (unit_id);

CREATE OR REPLACE FUNCTION products_validate_catalog_references()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_category_status catalog_visibility_status;
    v_unit_active boolean;
    v_allows_fractional boolean;
    v_decimal_precision smallint;
BEGIN
    SELECT status
    INTO v_category_status
    FROM product_categories
    WHERE id = NEW.category_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'category % does not exist', NEW.category_id;
    END IF;

    IF NEW.status = 'show' AND v_category_status <> 'show' THEN
        RAISE EXCEPTION 'visible product requires a visible category';
    END IF;

    SELECT is_active, allows_fractional, decimal_precision
    INTO v_unit_active, v_allows_fractional, v_decimal_precision
    FROM measurement_units
    WHERE id = NEW.unit_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'measurement unit % does not exist', NEW.unit_id;
    END IF;

    IF v_unit_active = false THEN
        RAISE EXCEPTION 'measurement unit % is inactive', NEW.unit_id;
    END IF;

    IF v_allows_fractional = false
        AND (
            NEW.min_order_quantity <> trunc(NEW.min_order_quantity)
            OR NEW.quantity_step <> trunc(NEW.quantity_step)
            OR (NEW.max_order_quantity IS NOT NULL AND NEW.max_order_quantity <> trunc(NEW.max_order_quantity))
            OR (NEW.max_daily_limit IS NOT NULL AND NEW.max_daily_limit <> trunc(NEW.max_daily_limit))
        )
    THEN
        RAISE EXCEPTION 'unit does not allow fractional product quantities';
    END IF;

    IF v_allows_fractional = true
        AND (
            NEW.min_order_quantity <> round(NEW.min_order_quantity, v_decimal_precision::integer)
            OR NEW.quantity_step <> round(NEW.quantity_step, v_decimal_precision::integer)
            OR (NEW.max_order_quantity IS NOT NULL AND NEW.max_order_quantity <> round(NEW.max_order_quantity, v_decimal_precision::integer))
            OR (NEW.max_daily_limit IS NOT NULL AND NEW.max_daily_limit <> round(NEW.max_daily_limit, v_decimal_precision::integer))
        )
    THEN
        RAISE EXCEPTION 'product quantity precision exceeds unit precision';
    END IF;

    RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_products_validate_catalog_references ON products;

CREATE TRIGGER trg_products_validate_catalog_references
BEFORE INSERT OR UPDATE OF category_id, unit_id, status, min_order_quantity, max_order_quantity, quantity_step, max_daily_limit ON products
FOR EACH ROW
EXECUTE FUNCTION products_validate_catalog_references();

DROP TRIGGER IF EXISTS trg_products_set_updated_at ON products;

CREATE TRIGGER trg_products_set_updated_at
BEFORE UPDATE ON products
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_products_outbox ON products;

CREATE TRIGGER trg_products_outbox
AFTER INSERT OR UPDATE OR DELETE ON products
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
