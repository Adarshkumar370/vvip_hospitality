-- Functions to check if a user can place an order.
-- Use can_user_place_order() in the API and in the orders trigger.

CREATE OR REPLACE FUNCTION can_user_place_order(p_user_id uuid)
RETURNS boolean
LANGUAGE sql
STABLE
AS $$
    SELECT EXISTS (
        SELECT 1
        FROM app_users u
        WHERE u.id = p_user_id
          AND u.status = 'allowed'
          AND u.mobile_no IS NOT NULL
          AND u.email IS NOT NULL
          AND u.mobile_verified_at IS NOT NULL
          AND u.email_verified_at IS NOT NULL
    );
$$;

CREATE OR REPLACE FUNCTION get_user_order_block_reason(p_user_id uuid)
RETURNS text
LANGUAGE plpgsql
STABLE
AS $$
DECLARE
    v_user app_users%ROWTYPE;
BEGIN
    SELECT *
    INTO v_user
    FROM app_users
    WHERE id = p_user_id;

    IF NOT FOUND THEN
        RETURN 'user_not_found';
    END IF;

    IF v_user.status <> 'allowed' THEN
        RETURN 'user_not_allowed';
    END IF;

    IF v_user.mobile_no IS NULL THEN
        RETURN 'mobile_missing';
    END IF;

    IF v_user.email IS NULL THEN
        RETURN 'email_missing';
    END IF;

    IF v_user.mobile_verified_at IS NULL THEN
        RETURN 'mobile_not_verified';
    END IF;

    IF v_user.email_verified_at IS NULL THEN
        RETURN 'email_not_verified';
    END IF;

    RETURN NULL;
END;
$$;
