-- Permission catalog and staff permission assignments.
-- Admin staff members are treated as having all active permissions. Other staff
-- need an active row in staff_permissions.

CREATE TABLE IF NOT EXISTS permissions (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    permission_code varchar(80) NOT NULL,
    display_name varchar(120) NOT NULL,
    description text,
    is_active boolean NOT NULL DEFAULT true,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_permissions_code
        CHECK (permission_code ~ '^[a-z][a-z0-9_]{1,79}$'),
    CONSTRAINT chk_permissions_display_name
        CHECK (length(btrim(display_name)) > 0),
    CONSTRAINT chk_permissions_description
        CHECK (description IS NULL OR length(btrim(description)) > 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_permissions_code
    ON permissions (permission_code);

CREATE TABLE IF NOT EXISTS staff_permissions (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    staff_id uuid NOT NULL REFERENCES staff_members(id) ON DELETE CASCADE,
    permission_id uuid NOT NULL REFERENCES permissions(id) ON DELETE RESTRICT,
    granted_by_staff_id uuid REFERENCES staff_members(id) ON DELETE SET NULL,
    grant_reason text,
    granted_at timestamptz NOT NULL DEFAULT now(),
    revoked_at timestamptz,
    revoked_by_staff_id uuid REFERENCES staff_members(id) ON DELETE SET NULL,
    revoke_reason text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_staff_permissions_grant_reason
        CHECK (grant_reason IS NULL OR length(btrim(grant_reason)) > 0),
    CONSTRAINT chk_staff_permissions_revoke_reason
        CHECK (revoke_reason IS NULL OR length(btrim(revoke_reason)) > 0),
    CONSTRAINT chk_staff_permissions_revoke_staff
        CHECK (
            (revoked_at IS NULL AND revoked_by_staff_id IS NULL)
            OR (revoked_at IS NOT NULL)
        )
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_staff_permissions_active
    ON staff_permissions (staff_id, permission_id)
    WHERE revoked_at IS NULL;

CREATE INDEX IF NOT EXISTS ix_staff_permissions_staff_id
    ON staff_permissions (staff_id);

CREATE INDEX IF NOT EXISTS ix_staff_permissions_permission_id
    ON staff_permissions (permission_id);

CREATE OR REPLACE FUNCTION staff_has_permission(
    p_staff_id uuid,
    p_permission_code varchar
)
RETURNS boolean
LANGUAGE sql
STABLE
AS $$
    SELECT EXISTS (
        SELECT 1
        FROM staff_members sm
        WHERE sm.id = p_staff_id
          AND sm.status = 'active'
          AND sm.designation = 'admin'
    )
    OR EXISTS (
        SELECT 1
        FROM staff_permissions sp
        JOIN staff_members sm ON sm.id = sp.staff_id
        JOIN permissions p ON p.id = sp.permission_id
        WHERE sp.staff_id = p_staff_id
          AND sm.status = 'active'
          AND p.is_active = true
          AND p.permission_code = p_permission_code
          AND sp.revoked_at IS NULL
    );
$$;

DROP TRIGGER IF EXISTS trg_permissions_set_updated_at ON permissions;

CREATE TRIGGER trg_permissions_set_updated_at
BEFORE UPDATE ON permissions
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_permissions_outbox ON permissions;

CREATE TRIGGER trg_permissions_outbox
AFTER INSERT OR UPDATE OR DELETE ON permissions
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();

CREATE TRIGGER trg_staff_permissions_set_updated_at
BEFORE UPDATE ON staff_permissions
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_staff_permissions_outbox ON staff_permissions;

CREATE TRIGGER trg_staff_permissions_outbox
AFTER INSERT OR UPDATE OR DELETE ON staff_permissions
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
