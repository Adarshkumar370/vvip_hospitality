-- Measurement units used by products.
-- Products reference this table instead of storing a free-text unit value.

CREATE TABLE IF NOT EXISTS measurement_units (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    code varchar(30) NOT NULL,
    display_name varchar(80) NOT NULL,
    symbol varchar(20) NOT NULL,
    allows_fractional boolean NOT NULL DEFAULT false,
    decimal_precision smallint NOT NULL DEFAULT 0,
    is_active boolean NOT NULL DEFAULT true,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_measurement_units_code
        CHECK (code ~ '^[a-z][a-z0-9_]{1,29}$'),
    CONSTRAINT chk_measurement_units_display_name
        CHECK (length(btrim(display_name)) > 0),
    CONSTRAINT chk_measurement_units_symbol
        CHECK (length(btrim(symbol)) > 0),
    CONSTRAINT chk_measurement_units_decimal_precision
        CHECK (decimal_precision BETWEEN 0 AND 4),
    CONSTRAINT chk_measurement_units_fractional_precision
        CHECK (
            (allows_fractional = true AND decimal_precision > 0)
            OR (allows_fractional = false AND decimal_precision = 0)
        )
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_measurement_units_code
    ON measurement_units (code);

DROP TRIGGER IF EXISTS trg_measurement_units_set_updated_at ON measurement_units;

CREATE TRIGGER trg_measurement_units_set_updated_at
BEFORE UPDATE ON measurement_units
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_measurement_units_outbox ON measurement_units;

CREATE TRIGGER trg_measurement_units_outbox
AFTER INSERT OR UPDATE OR DELETE ON measurement_units
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();

INSERT INTO measurement_units (
    code,
    display_name,
    symbol,
    allows_fractional,
    decimal_precision
)
VALUES
    ('piece', 'Piece', 'pc', false, 0),
    ('box', 'Box', 'box', false, 0),
    ('slice', 'Slice', 'slice', false, 0),
    ('dozen', 'Dozen', 'dz', false, 0),
    ('gram', 'Gram', 'g', true, 3),
    ('kilogram', 'Kilogram', 'kg', true, 3)
ON CONFLICT (code) DO UPDATE
SET display_name = EXCLUDED.display_name,
    symbol = EXCLUDED.symbol,
    allows_fractional = EXCLUDED.allows_fractional,
    decimal_precision = EXCLUDED.decimal_precision,
    is_active = true
WHERE measurement_units.display_name IS DISTINCT FROM EXCLUDED.display_name
   OR measurement_units.symbol IS DISTINCT FROM EXCLUDED.symbol
   OR measurement_units.allows_fractional IS DISTINCT FROM EXCLUDED.allows_fractional
   OR measurement_units.decimal_precision IS DISTINCT FROM EXCLUDED.decimal_precision
   OR measurement_units.is_active IS DISTINCT FROM true;
