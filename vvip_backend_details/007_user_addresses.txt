-- User addresses.
-- A user can have many addresses, but only one active default address.

CREATE TABLE IF NOT EXISTS user_addresses (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE CASCADE,
    address_label varchar(50) NOT NULL DEFAULT 'home',
    recipient_name varchar(150),
    recipient_mobile_no varchar(20),
    line1 text NOT NULL,
    line2 text,
    landmark text,
    city varchar(100) NOT NULL,
    state varchar(100),
    postal_code varchar(20) NOT NULL,
    country_code char(2) NOT NULL DEFAULT 'IN',
    latitude numeric(10, 7),
    longitude numeric(10, 7),
    delivery_instructions text,
    is_default boolean NOT NULL DEFAULT false,
    is_active boolean NOT NULL DEFAULT true,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_user_addresses_label_not_blank
        CHECK (length(btrim(address_label)) > 0),
    CONSTRAINT chk_user_addresses_recipient_name_not_blank
        CHECK (recipient_name IS NULL OR length(btrim(recipient_name)) > 0),
    CONSTRAINT chk_user_addresses_recipient_mobile_format
        CHECK (
            recipient_mobile_no IS NULL
            OR recipient_mobile_no ~ '^\+[1-9][0-9]{7,14}$'
        ),
    CONSTRAINT chk_user_addresses_line1_not_blank
        CHECK (length(btrim(line1)) > 0),
    CONSTRAINT chk_user_addresses_city_not_blank
        CHECK (length(btrim(city)) > 0),
    CONSTRAINT chk_user_addresses_postal_code_not_blank
        CHECK (length(btrim(postal_code)) > 0),
    CONSTRAINT chk_user_addresses_country_code
        CHECK (country_code ~ '^[A-Z]{2}$'),
    CONSTRAINT chk_user_addresses_latitude
        CHECK (latitude IS NULL OR latitude BETWEEN -90 AND 90),
    CONSTRAINT chk_user_addresses_longitude
        CHECK (longitude IS NULL OR longitude BETWEEN -180 AND 180)
);

CREATE INDEX IF NOT EXISTS ix_user_addresses_user_id
    ON user_addresses (user_id);

CREATE INDEX IF NOT EXISTS ix_user_addresses_active
    ON user_addresses (user_id, is_active);

CREATE UNIQUE INDEX IF NOT EXISTS ux_user_addresses_one_default
    ON user_addresses (user_id)
    WHERE is_default = true AND is_active = true;

CREATE OR REPLACE FUNCTION user_addresses_prepare_default()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    IF NEW.is_active = false THEN
        NEW.is_default := false;
        RETURN NEW;
    END IF;

    IF NEW.is_default = false
        AND NOT EXISTS (
            SELECT 1
            FROM user_addresses
            WHERE user_id = NEW.user_id
              AND id <> NEW.id
              AND is_active = true
              AND is_default = true
        )
    THEN
        NEW.is_default := true;
    END IF;

    IF NEW.is_default = true THEN
        UPDATE user_addresses
        SET is_default = false,
            updated_at = now(),
            version = version + 1
        WHERE user_id = NEW.user_id
          AND id <> NEW.id
          AND is_active = true
          AND is_default = true;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION user_addresses_ensure_default_after_change()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_user_id uuid;
BEGIN
    v_user_id := COALESCE(NEW.user_id, OLD.user_id);

    IF NOT EXISTS (
        SELECT 1
        FROM user_addresses
        WHERE user_id = v_user_id
          AND is_active = true
          AND is_default = true
    )
    AND EXISTS (
        SELECT 1
        FROM user_addresses
        WHERE user_id = v_user_id
          AND is_active = true
    )
    THEN
        UPDATE user_addresses
        SET is_default = true
        WHERE id = (
            SELECT id
            FROM user_addresses
            WHERE user_id = v_user_id
              AND is_active = true
            ORDER BY updated_at DESC, created_at DESC
            LIMIT 1
        );
    END IF;

    RETURN COALESCE(NEW, OLD);
END;
$$;

DROP TRIGGER IF EXISTS trg_user_addresses_prepare_default ON user_addresses;

CREATE TRIGGER trg_user_addresses_prepare_default
BEFORE INSERT OR UPDATE OF user_id, is_default, is_active ON user_addresses
FOR EACH ROW
EXECUTE FUNCTION user_addresses_prepare_default();

DROP TRIGGER IF EXISTS trg_user_addresses_ensure_default_after_change ON user_addresses;

CREATE TRIGGER trg_user_addresses_ensure_default_after_change
AFTER INSERT OR UPDATE OR DELETE ON user_addresses
FOR EACH ROW
EXECUTE FUNCTION user_addresses_ensure_default_after_change();

DROP TRIGGER IF EXISTS trg_user_addresses_set_updated_at ON user_addresses;

CREATE TRIGGER trg_user_addresses_set_updated_at
BEFORE UPDATE ON user_addresses
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_user_addresses_outbox ON user_addresses;

CREATE TRIGGER trg_user_addresses_outbox
AFTER INSERT OR UPDATE OR DELETE ON user_addresses
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
