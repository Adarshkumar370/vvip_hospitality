-- Customer orders.
-- Order totals are recalculated from order_items. Invoices should later
-- snapshot the order/order_items data for billing.

CREATE SEQUENCE IF NOT EXISTS order_number_seq
    START WITH 1
    INCREMENT BY 1;

CREATE TABLE IF NOT EXISTS orders (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_number varchar(40) NOT NULL DEFAULT ('ORD-' || lpad(nextval('order_number_seq')::text, 10, '0')),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE RESTRICT,
    order_status order_status NOT NULL DEFAULT 'payment_pending',
    fulfillment_type order_fulfillment_type NOT NULL DEFAULT 'delivery',
    order_source order_source NOT NULL DEFAULT 'web',
    placed_by order_placed_by NOT NULL DEFAULT 'self',
    placed_by_staff_id uuid REFERENCES staff_members(id) ON DELETE RESTRICT,
    payment_type_snapshot user_payment_type NOT NULL,
    currency_code char(3) NOT NULL DEFAULT 'INR',
    customer_name varchar(201) NOT NULL,
    customer_email citext NOT NULL,
    customer_mobile_no varchar(20) NOT NULL,
    delivery_address_id uuid REFERENCES user_addresses(id) ON DELETE SET NULL,
    delivery_address_snapshot jsonb,
    subtotal_amount numeric(12, 2) NOT NULL DEFAULT 0,
    discount_amount numeric(12, 2) NOT NULL DEFAULT 0,
    tax_amount numeric(12, 2) NOT NULL DEFAULT 0,
    delivery_fee_amount numeric(12, 2) NOT NULL DEFAULT 0,
    adjustment_amount numeric(12, 2) NOT NULL DEFAULT 0,
    total_amount numeric(12, 2) NOT NULL DEFAULT 0,
    requested_fulfillment_at timestamptz,
    placed_at timestamptz,
    confirmed_at timestamptz,
    preparing_at timestamptz,
    ready_at timestamptz,
    out_for_delivery_at timestamptz,
    completed_at timestamptz,
    cancelled_at timestamptz,
    cancellation_reason text,
    placement_notes text,
    customer_notes text,
    employee_notes text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_orders_number
        CHECK (length(btrim(order_number)) > 0),
    CONSTRAINT chk_orders_currency
        CHECK (currency_code ~ '^[A-Z]{3}$'),
    CONSTRAINT chk_orders_customer_name
        CHECK (length(btrim(customer_name)) > 0),
    CONSTRAINT chk_orders_customer_email
        CHECK (length(btrim(customer_email::text)) > 0),
    CONSTRAINT chk_orders_customer_mobile_no
        CHECK (customer_mobile_no ~ '^\+[1-9][0-9]{7,14}$'),
    CONSTRAINT chk_orders_delivery_address
        CHECK (fulfillment_type <> 'delivery' OR delivery_address_snapshot IS NOT NULL),
    CONSTRAINT chk_orders_placed_by
        CHECK (
            (placed_by = 'self' AND placed_by_staff_id IS NULL)
            OR
            (
                placed_by = 'staff'
                AND placed_by_staff_id IS NOT NULL
                AND order_source IN ('phone', 'walk_in', 'employee', 'admin')
            )
        ),
    CONSTRAINT chk_orders_non_negative_amounts
        CHECK (
            subtotal_amount >= 0
            AND discount_amount >= 0
            AND tax_amount >= 0
            AND delivery_fee_amount >= 0
            AND total_amount >= 0
        ),
    CONSTRAINT chk_orders_cancellation_reason
        CHECK (cancellation_reason IS NULL OR length(btrim(cancellation_reason)) > 0),
    CONSTRAINT chk_orders_placement_notes
        CHECK (placement_notes IS NULL OR length(btrim(placement_notes)) > 0),
    CONSTRAINT chk_orders_customer_notes
        CHECK (customer_notes IS NULL OR length(btrim(customer_notes)) > 0),
    CONSTRAINT chk_orders_employee_notes
        CHECK (employee_notes IS NULL OR length(btrim(employee_notes)) > 0),
    CONSTRAINT chk_orders_completed_at
        CHECK (
            (order_status = 'completed' AND completed_at IS NOT NULL)
            OR order_status <> 'completed'
        ),
    CONSTRAINT chk_orders_cancelled_at
        CHECK (
            (order_status = 'cancelled' AND cancelled_at IS NOT NULL)
            OR order_status <> 'cancelled'
        )
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_orders_order_number
    ON orders (order_number);

CREATE INDEX IF NOT EXISTS ix_orders_user_id
    ON orders (user_id, created_at DESC);

CREATE INDEX IF NOT EXISTS ix_orders_status
    ON orders (order_status, requested_fulfillment_at);

CREATE INDEX IF NOT EXISTS ix_orders_fulfillment
    ON orders (fulfillment_type, order_status, requested_fulfillment_at);

CREATE INDEX IF NOT EXISTS ix_orders_placed_by_staff
    ON orders (placed_by_staff_id, created_at DESC)
    WHERE placed_by_staff_id IS NOT NULL;

CREATE OR REPLACE FUNCTION orders_prepare_user_snapshot()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_user app_users%ROWTYPE;
    v_address user_addresses%ROWTYPE;
    v_staff staff_members%ROWTYPE;
    v_delivery_address_id uuid;
    v_block_reason text;
BEGIN
    IF NOT can_user_place_order(NEW.user_id) THEN
        v_block_reason := get_user_order_block_reason(NEW.user_id);
        RAISE EXCEPTION 'user cannot place order: %', v_block_reason;
    END IF;

    SELECT *
    INTO v_user
    FROM app_users
    WHERE id = NEW.user_id;

    IF NEW.placed_by = 'self' THEN
        NEW.placed_by_staff_id := NULL;
    ELSE
        IF NEW.placed_by_staff_id IS NULL THEN
            RAISE EXCEPTION 'placed_by_staff_id is required when placed_by = staff';
        END IF;

        SELECT *
        INTO v_staff
        FROM staff_members
        WHERE id = NEW.placed_by_staff_id;

        IF NOT FOUND OR v_staff.status <> 'active' THEN
            RAISE EXCEPTION 'staff member placing the order must be active';
        END IF;

        IF NOT staff_has_permission(NEW.placed_by_staff_id, 'place_order_on_behalf') THEN
            RAISE EXCEPTION 'staff member does not have permission to place orders on behalf of users';
        END IF;

        IF NEW.order_source = 'web' THEN
            NEW.order_source := 'phone';
        END IF;
    END IF;

    NEW.payment_type_snapshot := COALESCE(NEW.payment_type_snapshot, v_user.payment_type);
    NEW.customer_name := COALESCE(
        NULLIF(btrim(NEW.customer_name), ''),
        btrim(v_user.first_name || ' ' || COALESCE(v_user.last_name, ''))
    );
    NEW.customer_email := COALESCE(NEW.customer_email, v_user.email);
    NEW.customer_mobile_no := COALESCE(NEW.customer_mobile_no, v_user.mobile_no);

    IF NEW.fulfillment_type = 'delivery' THEN
        v_delivery_address_id := NEW.delivery_address_id;

        IF v_delivery_address_id IS NULL THEN
            SELECT id
            INTO v_delivery_address_id
            FROM user_addresses
            WHERE user_id = NEW.user_id
              AND is_active = true
              AND is_default = true
            LIMIT 1;
        END IF;

        SELECT *
        INTO v_address
        FROM user_addresses
        WHERE id = v_delivery_address_id;

        IF NOT FOUND OR v_address.user_id <> NEW.user_id OR v_address.is_active = false THEN
            RAISE EXCEPTION 'active delivery address is required for delivery orders';
        END IF;

        NEW.delivery_address_id := v_delivery_address_id;
        NEW.delivery_address_snapshot := COALESCE(
            NEW.delivery_address_snapshot,
            jsonb_build_object(
                'id', v_address.id,
                'address_label', v_address.address_label,
                'recipient_name', v_address.recipient_name,
                'recipient_mobile_no', v_address.recipient_mobile_no,
                'line1', v_address.line1,
                'line2', v_address.line2,
                'landmark', v_address.landmark,
                'city', v_address.city,
                'state', v_address.state,
                'postal_code', v_address.postal_code,
                'country_code', v_address.country_code,
                'latitude', v_address.latitude,
                'longitude', v_address.longitude,
                'delivery_instructions', v_address.delivery_instructions,
                'address_version', v_address.version
            )
        );
    ELSE
        NEW.delivery_address_id := NULL;
        NEW.delivery_address_snapshot := NULL;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION orders_prepare_amounts_and_timestamps()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.total_amount := round(
        NEW.subtotal_amount
        - NEW.discount_amount
        + NEW.tax_amount
        + NEW.delivery_fee_amount
        + NEW.adjustment_amount,
        2
    );

    IF NEW.total_amount < 0 THEN
        RAISE EXCEPTION 'order total cannot be negative';
    END IF;

    IF NEW.order_status IN (
        'payment_pending',
        'payment_failed',
        'placed',
        'confirmed',
        'preparing',
        'ready_for_pickup',
        'out_for_delivery',
        'completed'
    ) THEN
        NEW.placed_at := COALESCE(NEW.placed_at, now());
    END IF;

    IF NEW.order_status IN ('confirmed', 'preparing', 'ready_for_pickup', 'out_for_delivery', 'completed') THEN
        NEW.confirmed_at := COALESCE(NEW.confirmed_at, now());
    END IF;

    IF NEW.order_status IN ('preparing', 'ready_for_pickup', 'out_for_delivery', 'completed') THEN
        NEW.preparing_at := COALESCE(NEW.preparing_at, now());
    END IF;

    IF NEW.order_status IN ('ready_for_pickup', 'out_for_delivery', 'completed') THEN
        NEW.ready_at := COALESCE(NEW.ready_at, now());
    END IF;

    IF NEW.order_status IN ('out_for_delivery', 'completed') AND NEW.fulfillment_type = 'delivery' THEN
        NEW.out_for_delivery_at := COALESCE(NEW.out_for_delivery_at, now());
    END IF;

    IF NEW.order_status = 'completed' THEN
        NEW.completed_at := COALESCE(NEW.completed_at, now());
    ELSIF NEW.order_status = 'cancelled' THEN
        NEW.cancelled_at := COALESCE(NEW.cancelled_at, now());
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION orders_prevent_delete()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'orders cannot be deleted; cancel the order instead';
END;
$$;

DROP TRIGGER IF EXISTS trg_orders_prepare_user_snapshot ON orders;

CREATE TRIGGER trg_orders_prepare_user_snapshot
BEFORE INSERT OR UPDATE OF user_id, fulfillment_type, delivery_address_id, order_source, placed_by, placed_by_staff_id
ON orders
FOR EACH ROW
EXECUTE FUNCTION orders_prepare_user_snapshot();

DROP TRIGGER IF EXISTS trg_orders_prepare_amounts_and_timestamps ON orders;

CREATE TRIGGER trg_orders_prepare_amounts_and_timestamps
BEFORE INSERT OR UPDATE OF order_status, subtotal_amount, discount_amount, tax_amount, delivery_fee_amount, adjustment_amount
ON orders
FOR EACH ROW
EXECUTE FUNCTION orders_prepare_amounts_and_timestamps();

DROP TRIGGER IF EXISTS trg_orders_set_updated_at ON orders;

CREATE TRIGGER trg_orders_set_updated_at
BEFORE UPDATE ON orders
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_orders_prevent_delete ON orders;

CREATE TRIGGER trg_orders_prevent_delete
BEFORE DELETE ON orders
FOR EACH ROW
EXECUTE FUNCTION orders_prevent_delete();

DROP TRIGGER IF EXISTS trg_orders_outbox ON orders;

CREATE TRIGGER trg_orders_outbox
AFTER INSERT OR UPDATE OR DELETE ON orders
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
