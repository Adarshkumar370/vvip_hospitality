-- Shared trigger functions used by the bakery service schema.

CREATE OR REPLACE FUNCTION set_updated_at_and_version()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.updated_at := now();
    NEW.version := COALESCE(OLD.version, 0) + 1;
    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION publish_row_change_to_outbox()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_aggregate_id uuid;
    v_payload jsonb;
BEGIN
    IF TG_OP = 'DELETE' THEN
        v_aggregate_id := OLD.id;
        v_payload := jsonb_build_object(
            'operation', TG_OP,
            'old', to_jsonb(OLD)
        );
    ELSE
        v_aggregate_id := NEW.id;
        v_payload := jsonb_build_object(
            'operation', TG_OP,
            'new', to_jsonb(NEW),
            'old', CASE WHEN TG_OP = 'UPDATE' THEN to_jsonb(OLD) ELSE NULL END
        );
    END IF;

    INSERT INTO outbox_events (
        aggregate_type,
        aggregate_id,
        event_type,
        payload
    )
    VALUES (
        TG_TABLE_NAME,
        v_aggregate_id,
        TG_TABLE_NAME || '.' || lower(TG_OP),
        v_payload
    );

    RETURN COALESCE(NEW, OLD);
END;
$$;
