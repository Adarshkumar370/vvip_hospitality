-- Order line items.
-- Unit price is resolved from user_product_prices when present, otherwise from
-- products.price. The resolved values are stored as snapshots.

CREATE TABLE IF NOT EXISTS order_items (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id uuid NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    product_id uuid NOT NULL REFERENCES products(id) ON DELETE RESTRICT,
    category_id uuid NOT NULL REFERENCES product_categories(id) ON DELETE RESTRICT,
    unit_id uuid NOT NULL REFERENCES measurement_units(id) ON DELETE RESTRICT,
    line_number integer NOT NULL DEFAULT 1,
    product_name_snapshot text NOT NULL,
    category_name_snapshot text NOT NULL,
    category_slug_snapshot varchar(120) NOT NULL,
    unit_code_snapshot varchar(30) NOT NULL,
    unit_symbol_snapshot varchar(20) NOT NULL,
    quantity numeric(12, 3) NOT NULL,
    default_unit_price_snapshot numeric(12, 2) NOT NULL,
    custom_unit_price_snapshot numeric(12, 2),
    unit_price numeric(12, 2) NOT NULL,
    price_source order_price_source NOT NULL,
    line_subtotal numeric(12, 2) GENERATED ALWAYS AS (round(quantity * unit_price, 2)) STORED,
    discount_amount numeric(12, 2) NOT NULL DEFAULT 0,
    tax_amount numeric(12, 2) NOT NULL DEFAULT 0,
    line_total numeric(12, 2) GENERATED ALWAYS AS (
        round((quantity * unit_price) - discount_amount + tax_amount, 2)
    ) STORED,
    special_instructions text,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_order_items_line_number
        CHECK (line_number > 0),
    CONSTRAINT chk_order_items_product_name
        CHECK (length(btrim(product_name_snapshot)) > 0),
    CONSTRAINT chk_order_items_category_name
        CHECK (length(btrim(category_name_snapshot)) > 0),
    CONSTRAINT chk_order_items_category_slug
        CHECK (category_slug_snapshot ~ '^[a-z0-9]+(?:-[a-z0-9]+)*$'),
    CONSTRAINT chk_order_items_unit_code
        CHECK (unit_code_snapshot ~ '^[a-z][a-z0-9_]{1,29}$'),
    CONSTRAINT chk_order_items_unit_symbol
        CHECK (length(btrim(unit_symbol_snapshot)) > 0),
    CONSTRAINT chk_order_items_quantity
        CHECK (quantity > 0),
    CONSTRAINT chk_order_items_prices
        CHECK (
            default_unit_price_snapshot >= 0
            AND (custom_unit_price_snapshot IS NULL OR custom_unit_price_snapshot >= 0)
            AND unit_price >= 0
        ),
    CONSTRAINT chk_order_items_amounts
        CHECK (discount_amount >= 0 AND tax_amount >= 0 AND line_total >= 0),
    CONSTRAINT chk_order_items_special_instructions
        CHECK (special_instructions IS NULL OR length(btrim(special_instructions)) > 0)
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_order_items_order_line
    ON order_items (order_id, line_number);

CREATE INDEX IF NOT EXISTS ix_order_items_order_id
    ON order_items (order_id);

CREATE INDEX IF NOT EXISTS ix_order_items_product_id
    ON order_items (product_id);

CREATE INDEX IF NOT EXISTS ix_order_items_category_id
    ON order_items (category_id);

CREATE OR REPLACE FUNCTION order_items_prepare_snapshot()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_order_user_id uuid;
    v_product record;
    v_custom_price numeric(12, 2);
BEGIN
    SELECT user_id
    INTO v_order_user_id
    FROM orders
    WHERE id = NEW.order_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'order % does not exist', NEW.order_id;
    END IF;

    SELECT
        p.name AS product_name,
        p.price,
        p.status AS product_status,
        p.min_order_quantity,
        p.max_order_quantity,
        p.quantity_step,
        p.max_daily_limit,
        c.id AS category_id,
        c.name AS category_name,
        c.slug AS category_slug,
        c.status AS category_status,
        mu.id AS unit_id,
        mu.code AS unit_code,
        mu.symbol AS unit_symbol
    INTO v_product
    FROM products p
    JOIN product_categories c ON c.id = p.category_id
    JOIN measurement_units mu ON mu.id = p.unit_id
    WHERE p.id = NEW.product_id;

    IF NOT FOUND THEN
        RAISE EXCEPTION 'product % does not exist', NEW.product_id;
    END IF;

    IF v_product.product_status <> 'show' OR v_product.category_status <> 'show' THEN
        RAISE EXCEPTION 'cannot order hidden product or hidden category';
    END IF;

    IF NEW.quantity < v_product.min_order_quantity THEN
        RAISE EXCEPTION 'quantity is below product minimum';
    END IF;

    IF v_product.max_order_quantity IS NOT NULL AND NEW.quantity > v_product.max_order_quantity THEN
        RAISE EXCEPTION 'quantity is above product maximum';
    END IF;

    IF mod(NEW.quantity - v_product.min_order_quantity, v_product.quantity_step) <> 0 THEN
        RAISE EXCEPTION 'quantity does not match product quantity step';
    END IF;

    IF TG_OP = 'INSERT'
        OR NEW.product_id IS DISTINCT FROM OLD.product_id
        OR NEW.order_id IS DISTINCT FROM OLD.order_id
    THEN
        SELECT price
        INTO v_custom_price
        FROM user_product_prices
        WHERE user_id = v_order_user_id
          AND product_id = NEW.product_id
          AND is_active = true;

        NEW.category_id := v_product.category_id;
        NEW.unit_id := v_product.unit_id;
        NEW.product_name_snapshot := v_product.product_name::text;
        NEW.category_name_snapshot := v_product.category_name::text;
        NEW.category_slug_snapshot := v_product.category_slug;
        NEW.unit_code_snapshot := v_product.unit_code;
        NEW.unit_symbol_snapshot := v_product.unit_symbol;
        NEW.default_unit_price_snapshot := v_product.price;
        NEW.custom_unit_price_snapshot := v_custom_price;
        NEW.unit_price := COALESCE(v_custom_price, v_product.price);
        NEW.price_source := CASE
            WHEN v_custom_price IS NULL THEN 'default_price'::order_price_source
            ELSE 'custom_user_price'::order_price_source
        END;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION recalculate_order_totals(p_order_id uuid)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
    UPDATE orders
    SET subtotal_amount = COALESCE((
            SELECT round(sum(line_subtotal), 2)
            FROM order_items
            WHERE order_id = p_order_id
        ), 0),
        discount_amount = COALESCE((
            SELECT round(sum(discount_amount), 2)
            FROM order_items
            WHERE order_id = p_order_id
        ), 0),
        tax_amount = COALESCE((
            SELECT round(sum(tax_amount), 2)
            FROM order_items
            WHERE order_id = p_order_id
        ), 0)
    WHERE id = p_order_id;
END;
$$;

CREATE OR REPLACE FUNCTION order_items_recalculate_order_totals()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    IF TG_OP = 'UPDATE' AND NEW.order_id IS DISTINCT FROM OLD.order_id THEN
        PERFORM recalculate_order_totals(OLD.order_id);
        PERFORM recalculate_order_totals(NEW.order_id);
        RETURN NEW;
    END IF;

    PERFORM recalculate_order_totals(COALESCE(NEW.order_id, OLD.order_id));
    RETURN COALESCE(NEW, OLD);
END;
$$;

DROP TRIGGER IF EXISTS trg_order_items_prepare_snapshot ON order_items;

CREATE TRIGGER trg_order_items_prepare_snapshot
BEFORE INSERT OR UPDATE OF order_id, product_id, quantity
ON order_items
FOR EACH ROW
EXECUTE FUNCTION order_items_prepare_snapshot();

DROP TRIGGER IF EXISTS trg_order_items_recalculate_order_totals ON order_items;

CREATE TRIGGER trg_order_items_recalculate_order_totals
AFTER INSERT OR UPDATE OR DELETE ON order_items
FOR EACH ROW
EXECUTE FUNCTION order_items_recalculate_order_totals();

DROP TRIGGER IF EXISTS trg_order_items_set_updated_at ON order_items;

CREATE TRIGGER trg_order_items_set_updated_at
BEFORE UPDATE ON order_items
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_order_items_outbox ON order_items;

CREATE TRIGGER trg_order_items_outbox
AFTER INSERT OR UPDATE OR DELETE ON order_items
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
