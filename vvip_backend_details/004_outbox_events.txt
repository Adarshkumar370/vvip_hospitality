-- Reliable change queue for syncing PostgreSQL changes to MongoDB.
-- The worker should process pending rows in created_at order and then mark
-- them processed only after MongoDB has acknowledged the write.

CREATE TABLE IF NOT EXISTS outbox_events (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    aggregate_type varchar(80) NOT NULL,
    aggregate_id uuid NOT NULL,
    event_type varchar(120) NOT NULL,
    payload jsonb NOT NULL,
    payload_hash text GENERATED ALWAYS AS (
        encode(digest(payload::text, 'sha256'), 'hex')
    ) STORED,
    status outbox_event_status NOT NULL DEFAULT 'pending',
    attempt_count integer NOT NULL DEFAULT 0,
    last_error text,
    available_at timestamptz NOT NULL DEFAULT now(),
    processed_at timestamptz,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_outbox_events_attempt_count
        CHECK (attempt_count >= 0),
    CONSTRAINT chk_outbox_events_processed_at
        CHECK (
            (status = 'processed' AND processed_at IS NOT NULL)
            OR (status <> 'processed')
        )
);

CREATE INDEX IF NOT EXISTS ix_outbox_events_pending
    ON outbox_events (available_at, created_at)
    WHERE status IN ('pending', 'failed');

CREATE INDEX IF NOT EXISTS ix_outbox_events_aggregate
    ON outbox_events (aggregate_type, aggregate_id, created_at);

DROP TRIGGER IF EXISTS trg_outbox_events_set_updated_at ON outbox_events;

CREATE TRIGGER trg_outbox_events_set_updated_at
BEFORE UPDATE ON outbox_events
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();
