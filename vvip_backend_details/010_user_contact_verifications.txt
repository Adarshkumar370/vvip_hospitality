-- Stores email and mobile verification attempts.
-- Never store raw OTP values. Store only otp_hash or external provider refs.

CREATE TABLE IF NOT EXISTS user_contact_verifications (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE CASCADE,
    contact_type contact_type NOT NULL,
    contact_value text NOT NULL,
    verification_status verification_status NOT NULL DEFAULT 'pending',
    verification_channel varchar(30) NOT NULL,
    purpose varchar(40) NOT NULL DEFAULT 'signup',
    otp_hash text,
    attempt_count integer NOT NULL DEFAULT 0,
    max_attempts integer NOT NULL DEFAULT 5,
    sent_at timestamptz NOT NULL DEFAULT now(),
    expires_at timestamptz NOT NULL,
    verified_at timestamptz,
    provider varchar(80),
    provider_reference varchar(160),
    idempotency_key varchar(160),
    request_id uuid,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_user_contact_verifications_contact_value
        CHECK (length(btrim(contact_value)) > 0),
    CONSTRAINT chk_user_contact_verifications_contact_format
        CHECK (
            (contact_type = 'mobile' AND contact_value ~ '^\+[1-9][0-9]{7,14}$')
            OR
            (contact_type = 'email' AND contact_value LIKE '%@%')
        ),
    CONSTRAINT chk_user_contact_verifications_channel
        CHECK (verification_channel IN ('sms', 'whatsapp', 'email')),
    CONSTRAINT chk_user_contact_verifications_attempts
        CHECK (attempt_count >= 0 AND max_attempts BETWEEN 1 AND 20 AND attempt_count <= max_attempts),
    CONSTRAINT chk_user_contact_verifications_expiry
        CHECK (expires_at > sent_at),
    CONSTRAINT chk_user_contact_verifications_verified_at
        CHECK (
            (verification_status = 'verified' AND verified_at IS NOT NULL)
            OR (verification_status <> 'verified')
        )
);

CREATE INDEX IF NOT EXISTS ix_user_contact_verifications_user_id
    ON user_contact_verifications (user_id, created_at DESC);

CREATE INDEX IF NOT EXISTS ix_user_contact_verifications_pending
    ON user_contact_verifications (expires_at)
    WHERE verification_status = 'pending';

CREATE UNIQUE INDEX IF NOT EXISTS ux_user_contact_verifications_pending_contact
    ON user_contact_verifications (user_id, contact_type, contact_value)
    WHERE verification_status = 'pending';

CREATE UNIQUE INDEX IF NOT EXISTS ux_user_contact_verifications_idempotency
    ON user_contact_verifications (user_id, contact_type, idempotency_key)
    WHERE idempotency_key IS NOT NULL;

CREATE OR REPLACE FUNCTION user_contact_verifications_apply_success()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_matched_count integer;
BEGIN
    IF NEW.verification_status <> 'verified' THEN
        RETURN NEW;
    END IF;

    IF TG_OP = 'UPDATE'
        AND OLD.verification_status = 'verified'
        AND OLD.verified_at IS NOT DISTINCT FROM NEW.verified_at
    THEN
        RETURN NEW;
    END IF;

    IF NEW.contact_type = 'email' THEN
        UPDATE app_users
        SET email_verified_at = COALESCE(email_verified_at, NEW.verified_at)
        WHERE id = NEW.user_id
          AND email = NEW.contact_value::citext;
    ELSE
        UPDATE app_users
        SET mobile_verified_at = COALESCE(mobile_verified_at, NEW.verified_at)
        WHERE id = NEW.user_id
          AND mobile_no = NEW.contact_value;
    END IF;

    GET DIAGNOSTICS v_matched_count = ROW_COUNT;

    IF v_matched_count = 0 THEN
        RAISE EXCEPTION 'verified contact does not match current app_users contact for user_id %', NEW.user_id;
    END IF;

    RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_user_contact_verifications_apply_success ON user_contact_verifications;

CREATE TRIGGER trg_user_contact_verifications_apply_success
AFTER INSERT OR UPDATE OF verification_status, verified_at ON user_contact_verifications
FOR EACH ROW
EXECUTE FUNCTION user_contact_verifications_apply_success();

DROP TRIGGER IF EXISTS trg_user_contact_verifications_set_updated_at ON user_contact_verifications;

CREATE TRIGGER trg_user_contact_verifications_set_updated_at
BEFORE UPDATE ON user_contact_verifications
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_user_contact_verifications_outbox ON user_contact_verifications;

CREATE TRIGGER trg_user_contact_verifications_outbox
AFTER INSERT OR UPDATE OR DELETE ON user_contact_verifications
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
