-- Enum definitions used by the order module.

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'order_status') THEN
        CREATE TYPE order_status AS ENUM (
            'payment_pending',
            'payment_failed',
            'placed',
            'confirmed',
            'preparing',
            'ready_for_pickup',
            'out_for_delivery',
            'completed',
            'cancelled'
        );
    END IF;
END
$$;

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'order_fulfillment_type') THEN
        CREATE TYPE order_fulfillment_type AS ENUM ('pickup', 'delivery');
    END IF;
END
$$;

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'order_source') THEN
        CREATE TYPE order_source AS ENUM ('web', 'phone', 'walk_in', 'employee', 'admin');
    END IF;
END
$$;

ALTER TYPE order_source ADD VALUE IF NOT EXISTS 'phone';
ALTER TYPE order_source ADD VALUE IF NOT EXISTS 'walk_in';

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'order_placed_by') THEN
        CREATE TYPE order_placed_by AS ENUM ('self', 'staff');
    END IF;
END
$$;

DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'order_price_source') THEN
        CREATE TYPE order_price_source AS ENUM ('default_price', 'custom_user_price');
    END IF;
END
$$;
