-- Payment attempts and transactions.
-- Pending and failed payment data belongs here. Do not delete payment rows;
-- update status and keep provider/failure details for audit and reconciliation.

CREATE TABLE IF NOT EXISTS payments (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE RESTRICT,
    invoice_id uuid REFERENCES invoices(id) ON DELETE RESTRICT,
    order_id uuid REFERENCES orders(id) ON DELETE RESTRICT,
    amount numeric(12, 2) NOT NULL,
    currency_code char(3) NOT NULL DEFAULT 'INR',
    payment_method payment_method NOT NULL,
    payment_status payment_status NOT NULL DEFAULT 'pending',
    payment_provider varchar(80),
    provider_payment_id varchar(160),
    provider_order_id varchar(160),
    provider_status varchar(80),
    provider_status_message text,
    failure_code varchar(120),
    failure_message text,
    idempotency_key varchar(160),
    request_hash text,
    provider_request_payload jsonb,
    provider_response_payload jsonb,
    initiated_at timestamptz NOT NULL DEFAULT now(),
    authorized_at timestamptz,
    captured_at timestamptz,
    failed_at timestamptz,
    cancelled_at timestamptz,
    expires_at timestamptz,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_payments_invoice_or_order
        CHECK (invoice_id IS NOT NULL OR order_id IS NOT NULL),
    CONSTRAINT chk_payments_amount
        CHECK (amount > 0),
    CONSTRAINT chk_payments_currency
        CHECK (currency_code ~ '^[A-Z]{3}$'),
    CONSTRAINT chk_payments_provider
        CHECK (payment_provider IS NULL OR length(btrim(payment_provider)) > 0),
    CONSTRAINT chk_payments_provider_payment_id
        CHECK (provider_payment_id IS NULL OR length(btrim(provider_payment_id)) > 0),
    CONSTRAINT chk_payments_provider_order_id
        CHECK (provider_order_id IS NULL OR length(btrim(provider_order_id)) > 0),
    CONSTRAINT chk_payments_provider_status_message
        CHECK (provider_status_message IS NULL OR length(btrim(provider_status_message)) > 0),
    CONSTRAINT chk_payments_failure_code
        CHECK (failure_code IS NULL OR length(btrim(failure_code)) > 0),
    CONSTRAINT chk_payments_failure_message
        CHECK (failure_message IS NULL OR length(btrim(failure_message)) > 0),
    CONSTRAINT chk_payments_idempotency_key
        CHECK (idempotency_key IS NULL OR length(btrim(idempotency_key)) > 0),
    CONSTRAINT chk_payments_request_hash
        CHECK (request_hash IS NULL OR length(btrim(request_hash)) > 0),
    CONSTRAINT chk_payments_failed_at
        CHECK (
            (payment_status = 'failed' AND failed_at IS NOT NULL)
            OR payment_status <> 'failed'
        ),
    CONSTRAINT chk_payments_captured_at
        CHECK (
            (payment_status = 'succeeded' AND captured_at IS NOT NULL)
            OR payment_status <> 'succeeded'
        ),
    CONSTRAINT chk_payments_cancelled_at
        CHECK (
            (payment_status = 'cancelled' AND cancelled_at IS NOT NULL)
            OR payment_status <> 'cancelled'
        )
);

CREATE INDEX IF NOT EXISTS ix_payments_user_id
    ON payments (user_id, created_at DESC);

CREATE INDEX IF NOT EXISTS ix_payments_invoice_id
    ON payments (invoice_id, created_at DESC)
    WHERE invoice_id IS NOT NULL;

CREATE INDEX IF NOT EXISTS ix_payments_order_id
    ON payments (order_id, created_at DESC)
    WHERE order_id IS NOT NULL;

CREATE INDEX IF NOT EXISTS ix_payments_pending_failed
    ON payments (payment_status, created_at DESC)
    WHERE payment_status IN ('pending', 'processing', 'failed');

CREATE UNIQUE INDEX IF NOT EXISTS ux_payments_provider_payment_id
    ON payments (payment_provider, provider_payment_id)
    WHERE payment_provider IS NOT NULL AND provider_payment_id IS NOT NULL;

CREATE UNIQUE INDEX IF NOT EXISTS ux_payments_provider_idempotency
    ON payments (payment_provider, idempotency_key)
    WHERE payment_provider IS NOT NULL AND idempotency_key IS NOT NULL;

CREATE OR REPLACE FUNCTION payments_prepare_status_timestamps()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    IF NEW.payment_status = 'authorized' AND NEW.authorized_at IS NULL THEN
        NEW.authorized_at := now();
    ELSIF NEW.payment_status = 'succeeded' AND NEW.captured_at IS NULL THEN
        NEW.captured_at := now();
    ELSIF NEW.payment_status = 'failed' AND NEW.failed_at IS NULL THEN
        NEW.failed_at := now();
    ELSIF NEW.payment_status = 'cancelled' AND NEW.cancelled_at IS NULL THEN
        NEW.cancelled_at := now();
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION payments_validate_payment_target()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_invoice_user_id uuid;
    v_invoice_order_id uuid;
    v_invoice_status invoice_status;
    v_order_user_id uuid;
    v_order_status order_status;
BEGIN
    IF NEW.invoice_id IS NOT NULL THEN
        SELECT user_id, order_id, invoice_status
        INTO v_invoice_user_id, v_invoice_order_id, v_invoice_status
        FROM invoices
        WHERE id = NEW.invoice_id;

        IF NOT FOUND THEN
            RAISE EXCEPTION 'invoice % does not exist', NEW.invoice_id;
        END IF;

        IF v_invoice_user_id <> NEW.user_id THEN
            RAISE EXCEPTION 'payment user_id must match invoice user_id';
        END IF;

        IF NEW.order_id IS NOT NULL AND v_invoice_order_id IS DISTINCT FROM NEW.order_id THEN
            RAISE EXCEPTION 'payment order_id must match invoice order_id';
        END IF;

        IF v_invoice_status IN ('void', 'cancelled') THEN
            RAISE EXCEPTION 'cannot create or update payment for invoice with status %', v_invoice_status;
        END IF;
    END IF;

    IF NEW.order_id IS NOT NULL THEN
        SELECT user_id, order_status
        INTO v_order_user_id, v_order_status
        FROM orders
        WHERE id = NEW.order_id;

        IF NOT FOUND THEN
            RAISE EXCEPTION 'order % does not exist', NEW.order_id;
        END IF;

        IF v_order_user_id <> NEW.user_id THEN
            RAISE EXCEPTION 'payment user_id must match order user_id';
        END IF;

        IF v_order_status = 'cancelled' THEN
            RAISE EXCEPTION 'cannot create or update payment for cancelled order';
        END IF;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION recalculate_invoice_paid_amount(p_invoice_id uuid)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
    IF p_invoice_id IS NULL THEN
        RETURN;
    END IF;

    UPDATE invoices
    SET paid_amount = COALESCE((
            SELECT round(sum(amount), 2)
            FROM payments
            WHERE invoice_id = p_invoice_id
              AND payment_status = 'succeeded'
        ), 0)
    WHERE id = p_invoice_id;
END;
$$;

CREATE OR REPLACE FUNCTION payments_recalculate_invoice_paid_amount()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    IF TG_OP = 'UPDATE' AND NEW.invoice_id IS DISTINCT FROM OLD.invoice_id THEN
        PERFORM recalculate_invoice_paid_amount(OLD.invoice_id);
        PERFORM recalculate_invoice_paid_amount(NEW.invoice_id);
        RETURN NEW;
    END IF;

    PERFORM recalculate_invoice_paid_amount(COALESCE(NEW.invoice_id, OLD.invoice_id));
    RETURN COALESCE(NEW, OLD);
END;
$$;

CREATE OR REPLACE FUNCTION payments_apply_order_payment_status()
RETURNS trigger
LANGUAGE plpgsql
AS $$
DECLARE
    v_order_id uuid;
BEGIN
    v_order_id := NEW.order_id;

    IF v_order_id IS NULL AND NEW.invoice_id IS NOT NULL THEN
        SELECT order_id
        INTO v_order_id
        FROM invoices
        WHERE id = NEW.invoice_id;
    END IF;

    IF v_order_id IS NULL THEN
        RETURN NEW;
    END IF;

    IF NEW.payment_status = 'succeeded' THEN
        UPDATE orders
        SET order_status = 'placed'
        WHERE id = v_order_id
          AND order_status IN ('payment_pending', 'payment_failed');
    ELSIF NEW.payment_status = 'failed' THEN
        UPDATE orders
        SET order_status = 'payment_failed'
        WHERE id = v_order_id
          AND order_status = 'payment_pending';
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION payments_prevent_delete()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE EXCEPTION 'payments are append/update-only and cannot be deleted';
END;
$$;

DROP TRIGGER IF EXISTS trg_payments_prepare_status_timestamps ON payments;

CREATE TRIGGER trg_payments_prepare_status_timestamps
BEFORE INSERT OR UPDATE OF payment_status ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_prepare_status_timestamps();

DROP TRIGGER IF EXISTS trg_payments_validate_payment_target ON payments;

CREATE TRIGGER trg_payments_validate_payment_target
BEFORE INSERT OR UPDATE OF user_id, invoice_id, order_id, payment_status ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_validate_payment_target();

DROP TRIGGER IF EXISTS trg_payments_recalculate_invoice_paid_amount ON payments;

CREATE TRIGGER trg_payments_recalculate_invoice_paid_amount
AFTER INSERT OR UPDATE OF payment_status, amount, invoice_id ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_recalculate_invoice_paid_amount();

DROP TRIGGER IF EXISTS trg_payments_apply_order_payment_status ON payments;

CREATE TRIGGER trg_payments_apply_order_payment_status
AFTER INSERT OR UPDATE OF payment_status, order_id, invoice_id ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_apply_order_payment_status();

DROP TRIGGER IF EXISTS trg_payments_prevent_delete ON payments;

CREATE TRIGGER trg_payments_prevent_delete
BEFORE DELETE ON payments
FOR EACH ROW
EXECUTE FUNCTION payments_prevent_delete();

DROP TRIGGER IF EXISTS trg_payments_set_updated_at ON payments;

CREATE TRIGGER trg_payments_set_updated_at
BEFORE UPDATE ON payments
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_payments_outbox ON payments;

CREATE TRIGGER trg_payments_outbox
AFTER INSERT OR UPDATE ON payments
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
