-- Optional per-user product price override.
-- products.price remains the default price. If no active row exists here for
-- the user/product pair, the application should use products.price.

CREATE TABLE IF NOT EXISTS user_product_prices (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id uuid NOT NULL REFERENCES app_users(id) ON DELETE CASCADE,
    product_id uuid NOT NULL REFERENCES products(id) ON DELETE CASCADE,
    price numeric(12, 2) NOT NULL,
    is_active boolean NOT NULL DEFAULT true,
    reason text,
    created_by_actor_type varchar(30) NOT NULL DEFAULT 'system',
    created_by_actor_id uuid,
    created_at timestamptz NOT NULL DEFAULT now(),
    updated_at timestamptz NOT NULL DEFAULT now(),
    version bigint NOT NULL DEFAULT 1,

    CONSTRAINT chk_user_product_prices_price
        CHECK (price >= 0),
    CONSTRAINT chk_user_product_prices_reason
        CHECK (reason IS NULL OR length(btrim(reason)) > 0),
    CONSTRAINT chk_user_product_prices_actor_type
        CHECK (created_by_actor_type IN ('system', 'employee'))
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_user_product_prices_user_product
    ON user_product_prices (user_id, product_id);

CREATE INDEX IF NOT EXISTS ix_user_product_prices_user_active
    ON user_product_prices (user_id, is_active, product_id);

CREATE INDEX IF NOT EXISTS ix_user_product_prices_product_id
    ON user_product_prices (product_id);

DROP TRIGGER IF EXISTS trg_user_product_prices_set_updated_at ON user_product_prices;

CREATE TRIGGER trg_user_product_prices_set_updated_at
BEFORE UPDATE ON user_product_prices
FOR EACH ROW
EXECUTE FUNCTION set_updated_at_and_version();

DROP TRIGGER IF EXISTS trg_user_product_prices_outbox ON user_product_prices;

CREATE TRIGGER trg_user_product_prices_outbox
AFTER INSERT OR UPDATE OR DELETE ON user_product_prices
FOR EACH ROW
EXECUTE FUNCTION publish_row_change_to_outbox();
